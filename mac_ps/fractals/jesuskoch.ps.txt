28.3464567 dup scale

newpath
0.5 0.5 moveto
20.5 0.5 lineto
20.5 29.2 lineto
0.5 29.2 lineto
closepath 
%gsave
clip
%grestore
%0 setgray
%fill

/height 29.7039 def
/width 20.9903 def

width 0.5 mul height 0.5 mul translate

0.02  setlinewidth

1 setgray

/koch {
6 dict begin

	/rotation exch def
	/lx exch def
	/ly exch def
	/length exch def
	/depth exch def

	/alt 3 sqrt 2 div length 3 div mul def

	lx ly translate rotation rotate

	newpath
	length 3 div 0 moveto
	length 2 div alt lineto
	length 3 div 2 mul 0 lineto
	closepath fill

	newpath

	depth 0 eq not {
		gsave
		depth 1 sub length 3 div 0 0 0 koch
		grestore gsave
		depth 1 sub length 3 div 0 0 length 3 div add 60 koch
		grestore gsave
		depth 1 sub length 3 div alt length 2 div 300 koch
		grestore gsave
		depth 1 sub length 3 div 0 2 length 3 div mul 0 koch
		grestore
	} if

end
} def

newpath
-8 5 moveto
8 5 lineto
0 5 16 3 sqrt 2 div mul sub lineto
closepath fill

/gd 4 def

/snowflake {
	gsave
	gd 16 5 -8 0 koch
	grestore gsave
	gd 16 5 16 3 sqrt 2 div mul sub 0 120 koch
	grestore gsave
	gd 16 5 8 240 koch
	grestore
} def

1 1 100 div neg 0 {
	setgray

	1 1 100 div add dup scale
	snowflake

} for


showpage
